import java.io.*;
import java.util.*;
class brute_force_1 
{ 
 public static void brute(String encText){ 
 encText = encText.toLowerCase();
 String text = "abcdefghijklmnopqrstuvwxyz";
 String finalValue="";
 for(int i=0;i< 26;i++){
 System.out.print("Key["+i+"]=");
 for(int j=0;j< encText.length();j++){
 char c = encText.charAt(j);
 int charInt = text.indexOf(c);
 int pos = (charInt-i)%26;
 if(pos< 0){ 
 pos = text.length()+pos;
 } 
 char encrChar = text.charAt(pos);
 finalValue+=encrChar;
 } 
 System.out.println(finalValue);
 finalValue="";
 } 
 } 
 public static void main(String args[]){ 
 Scanner sc = new Scanner(System.in);
 System.out.print("Enter The Encrypted Message :");
 String encText = sc.nextLine();
 brute(encText);
}}
