import java.util.Scanner;
public class decryption 
{ 
 public static void main(String[] args) 
 { 
 System.out.println("Ceaser cipher decryption\n");
 int key,i,index,e_index;
 String etext="",text;
String alphabet = "abcdefghijklmnopqrstuvwxyz";
 System.out.println("Enter the text to decrypt :");
 Scanner sc=new Scanner(System.in);
 text=sc.nextLine();
 System.out.println("Enter the key value:");
 key=sc.nextInt();
 for(i=0;i< text.length();i++)
 { 
 char current_char=text.charAt(i);
 index=alphabet.indexOf(current_char);
 e_index=(index-key)%26;
 etext=etext+alphabet.charAt(e_index);
 } 
 System.out.println("encrypted text is:"+etext);
 } 
}
